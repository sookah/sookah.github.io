# Inspects all variables used by a method
# and returns the method arguments
# leaving out 'self' if it is the first argument
@lru_cache(maxsize=16)
def get_method_args(method):
    args = method.__code__.co_varnames[:method.__code__.co_argcount]
    if args[0] == 'self':
        args = args[1:]
    return args


# Receives a psutil.Process object
# and serializes its selected attributes as a dict
# Attribute order is randomized by psutil.Process.as_dict()
# and should not be assumed for any functionality
def serialize_proc(proc: psutil.Process, args: list = ('name', 'exe',
                                                       'cmdline', 'pid',
                                                       'ppid', 'username',
                                                       'create_time', 'status',
                                                       'cpu_times', 'cpu_percent',
                                                       'open_files')):
    proc_attrs = set(attr for attr in dir(proc) if not attr.startswith('_'))
    if all(arg in proc_attrs for arg in args):
        serialized_proc = proc.as_dict(args)
    else:
        raise RuntimeError('Invalid psutil.Process attributes were passed to the function.')
    return serialized_proc


# Receives a psutil.Popen object
# and serializes its important info as a dict
def serialize_popen(popen: psutil.Popen):
    serialized_popen = {
        'name': popen.path,
        'mode': popen.mode,
    }
    return serialized_popen

[docs]    def get_proc(self, proc_name: str = None, proc_id: int = None, pproc_id: int = None, serialize=False):
        """get_proc(proc_name, proc_id, pproc_id, serialize)
        Gets a process from the list of currently running ones.

        :param proc_name: Name of the process to look up and perform the method for.\n
            *Must be a string unique to the name of a single process among the currently running ones
            if used without proc_id.*\n
            If unset, *proc_id* must be set instead.
            If both are set, *proc_id* is used for the lookup and *proc_name* is only used for verification.\n
            Internally uses :func:`get_procs_by_name` and checks if the returned list only contains a single item.
        :type proc_name: str
        :param proc_id: Process ID (PID) of the process to look up and perform the method for.\n
            If unset, *proc_name* must be set instead.
            If both are set, *proc_id* is used for the lookup and *proc_name* is only used for verification.\n
            If using *proc_id*, *pproc_id* can be used to indicate the parent process id
            in order to check for possible process id reuse.
        :type proc_id: int
        :param pproc_id: Expected Parent Process ID (PPID) for the process returned when looking up *proc_id*.\n
            Used to check for possible process id reuse, and only in conjunction with *proc_id*.
        :type pproc_id: int
        :param serialize: Tells the method whether to return a |psutil.Process| instance or serialize the
            important process info. This is intended to separate API returns (which would be better
            formated with a serialized object) versus internal usage which might require a |psutil.Process| object.\n
            Defaults to **False**.
        :type serialize: bool
        :return: Process whose attributes match the provided function arguments.
        :rtype: |psutil.Process| or dict
        """
        if not proc_name and not proc_id:
            raise Exception("Neither proc_name nor proc_id were provided, "
                            "so there is no process to perform the operation for.")
        if proc_id:
            if psutil.pid_exists(proc_id):
                proc = psutil.Process(proc_id)
                if pproc_id and proc.ppid() != pproc_id:
                    raise Exception(f'The provided parent process id {pproc_id} does not match '
                                    f'the parent process id of the process currently using '
                                    f'the process id {proc_id} (real ppid = {proc.ppid}).')
                if proc_name and proc_name not in proc.name():
                    raise Exception(f'A process with proc_id {proc_id} was found but '
                                    f'its name did not contaion {proc_name}.')

            else:
                raise Exception(f'No processs with the provided pid {proc_id} was found.')
        elif proc_name:
            lookup_result_procs = self.get_procs_by_name(proc_name)
            if len(lookup_result_procs) == 1:
                proc = lookup_result_procs[0]
            elif not lookup_result_procs:
                raise Exception(f'No processes with the specified string {proc_name} in their name were found.')
            elif len(lookup_result_procs) > 1:
                def unpack(procs):
                    return ', '.join(str(iter_proc.name()) for iter_proc in procs)
                raise Exception(f'Several processes with the specified string {proc_name} in their name were found. '
                                f'Discovered process names: {unpack(lookup_result_procs)}')
        if serialize:
            return serialize_proc(proc)

        return proc

    def get_procs_by_name(proc_name: str, serialize=False):
        """get_procs_by_name(proc_name, serialize)
        Gets a list of processes containing *proc_name* in their names.

        :param proc_name: Name of the process to look up.
        :type proc_name: str
        :param serialize: Tells the method whether to return a |psutil.Process| instance or serialize the
            important process info. This is intended to separate API returns (which would be better
            formated with a serialized object) versus internal usage which might require a |psutil.Process| object.\n
            Defaults to **False**.
        :type serialize: bool
        :return: Processes whose name contains *proc_name*.
        :rtype: list(|psutil.Process|) or list(dict)

        """
        result = []
        for proc in psutil.process_iter(attrs=["name"]):
            if proc_name in proc.info['name']:
                if serialize:
                    result.append(serialize_proc(proc))
                else:
                    result.append(proc)

        return result


[docs]    def kill_proc_tree(self, proc_name: str = None, proc_id: int = None,
                       pproc_id: int = None, sig: signal = signal.SIGTERM,
                       include_parent: bool = True, timeout: int = None,
                       serialize=False, on_terminate=None):
        """kill_proc_tree(proc_name, proc_id, pproc_id, sig, include_parent, timeout, serialize, on_terminate)
        Kills a process's tree, as well as the process itself if *include_parent* is **True**.

        :param proc_name: Name of the process to look up and perform the method for.\n
            *Must be a string unique to the name of a single process among the currently running ones
            if used without proc_id.*\n
            If unset, *proc_id* must be set instead.
            If both are set, *proc_id* is used for the lookup and *proc_name* is only used for verification.\n
            Internally uses :func:`get_procs_by_name` and checks if the returned list only contains a single item.
        :type proc_name: str
        :param proc_id: Process ID (PID) of the process to look up and perform the method for.\n
            If unset, *proc_name* must be set instead.
            If both are set, *proc_id* is used for the lookup and *proc_name* is only used for verification.\n
            If using *proc_id*, *pproc_id* can be used to indicate the parent process id
            in order to check for possible process id reuse.
        :type proc_id: int
        :param pproc_id: Expected Parent Process ID (PPID) for the process returned when looking up *proc_id*.\n
            Used to check for possible process id reuse, and only in conjunction with *proc_id*.
        :type pproc_id: int
        :param sig: Signal to send to all processes in the tree.
        :type sig: signal.SIGTERM
        :param include_parent: Indicates whether to send *signal* to the parent (*proc*) as well. Default to **True**.
        :type include_parent: bool
        :param timeout: Timeout to wait for process termination.
        :type timeout: int or None
        :param serialize: Tells the method whether to return a |psutil.Process| instance or serialize the
            important process info. This is intended to separate API returns (which would be better
            formated with a serialized object) versus internal usage which might require a |psutil.Process| object.\n
            Defaults to **False**.
        :type serialize: bool
        :param on_terminate: Function to execute upon termination of a process.
        :type on_terminate: function or None
        :return: Processes (|psutil.Process| or serialized to dict)
            that were terminated (*gone*) and processes that are still alive (*alive*).
        :rtype: :func:`python:collections.namedtuple`
        """
        # TODO: DON'T KILL MAIN APP EITHER (PLEASE)
        proc = self.get_proc(proc_name=proc_name, proc_id=proc_id, pproc_id=pproc_id)
        if proc == psutil.Process():
            raise RuntimeError("Can't tell the process to terminate itself")
        children = proc.children(recursive=True)
        if include_parent:
            children.append(proc)
        for child_proc in children:
            child_proc.send_signal(sig)
        result_state = namedtuple('result_state', ['gone', 'alive'])
        gone, alive = psutil.wait_procs(children, timeout=timeout, callback=on_terminate)
        if serialize:
            if gone and alive:
                return serialize_proc(gone), serialize_proc(alive)
            elif gone:
                return serialize_proc(gone)
            elif alive:
                return serialize_proc(alive)

        return result_state(gone=gone, alive=alive)

    def get_files_in_use():
        """get_files_in_use()
        Get all files currently being used. To be used when general info on all files
        curently open on the system is required. For info on a specific process,
        :func:`get_proc` is recommended instead.

        :return: Files currently open.
        :rtype: list(|psutil.Popen|)

        :Example:
            get_files_in_use() - list of all open files (|psutil.Popen|) across all processes\n
            get_files_in_use()[0] - process name and pid, as well as all open files (|psutil.Popen|) for said process\n
            get_files_in_use()[0]['open_files'] - list of all open files (|psutil.Popen|) for specific process\n
            get_files_in_use()[0]['open_files'][0] - info about specific open file (|psutil.Popen|)

        """
        result = []
        for proc in psutil.process_iter(attrs=['name', 'open_files']):
            if proc.info['open_files']:
                result.append({'name': proc.name(), 'pid': proc.pid, 'ppid': proc.ppid(),
                               'open_files': [serialize_popen(popen) for popen in proc.info['open_files']]})

        return result

                              proc_id: int = None, pproc_id: int = None):
        """get_bulk_process_data(proc_name, proc_id, pproc_id)
        Get all (relevant) process data in a single call.

        :param proc_name: Name of the process to look up and perform the method for.\n
            *Must be a string unique to the name of a single process among the currently running ones
            if used without proc_id.*\n
            If unset, *proc_id* must be set instead.
            If both are set, *proc_id* is used for the lookup and *proc_name* is only used for verification.\n
            Internally uses :func:`get_procs_by_name` and checks if the returned list only contains a single item.
        :type proc_name: str
        :param proc_id: Process ID (PID) of the process to look up and perform the method for.\n
            If unset, *proc_name* must be set instead.
            If both are set, *proc_id* is used for the lookup and *proc_name* is only used for verification.\n
            If using *proc_id*, *pproc_id* can be used to indicate the parent process id
            in order to check for possible process id reuse.
        :type proc_id: int
        :param pproc_id: Expected Parent Process ID (PPID) for the process returned when looking up *proc_id*.\n
            Used to check for possible process id reuse, and only in conjunction with *proc_id*.
        :type pproc_id: int
        :return: General process data - name, pid, owner (username), status,
            cpu time used since boot, and current cpu percentage.
        :rtype: dict

        """

        result = proc.as_dict(attrs=['name', 'ppid', 'username', 'status', 'cpu_times', 'cpu_percent'])

        return result


[docs]    @staticmethod
    def get_bulk_device_data():
        """get_bulk_device_data()
        Get all relevant device info.

        :return: General device data - CPU times and frequencies,
            disk usage statistics, boot time and currently logged users.
        :rtype: dict

        """
        result = {}
        result.update({'cpu_times': psutil.cpu_times()})
        result.update({'cpu_freq': psutil.cpu_freq()})
        result.update({'disk_usage': psutil.disk_usage('/')})
        result.update({'boot_time': datetime.fromtimestamp(psutil.boot_time()).strftime("%Y-%m-%d %H:%M:%S")})
        result.update({'users': psutil.users()})

        return result

        """get_open_ports(kind)

        :param kind: Connection type as defined
            `here <https://psutil.readthedocs.io/en/latest/#psutil.net_connections>`_. Defaults to 'inet'.

        open_ports = []

            open_ports.append(conn.laddr.port)

    def get_cpu_hogs(limit: int = 20, serialize=False):
        """get_cpu_hogs(limit, serialize)
        Gets the currently running processes that have used more than the specified amount of cpu time since booting.

        :param limit: |cpu_hog_limit_param|
        :type limit: int
        :param serialize: Tells the method whether to return a |psutil.Process| instance or serialize the
            important process info. This is intended to separate API returns (which would be better
            formated with a serialized object) versus internal usage which might require a |psutil.Process| object.\n
            Defaults to **False**.
        :type serialize: bool
        :return: Processes which took up more than the specified cpu time (*limit*) since booting.
        :rtype: list(|psutil.Process|) or list(dict)
        """
        proc_over_limit = []
        for proc in psutil.process_iter():
            total_proc_time = proc.cpu_times().system + proc.cpu_times().user
            if total_proc_time > limit:
                if serialize:
                    proc_over_limit.append(serialize_proc(proc))
                else:
                    proc_over_limit.append(proc)

        return proc_over_limit


[docs]    def change_cpu_hog_priority(self, limit: int = 20, niceness: int = 10):
        """change_cpu_hog_priority(limit, niceness)
        Changes the priority (niceness) of processes who've used over a certain amount of cpu time since booting,
        using :func:`get_cpu_hogs` internally to identify such processes.\n
        Niceness values range from -20 to 20 with higher values being lower priority; By default deprioritizes cpu hogs.

        :param limit: |cpu_hog_limit_param|
        :type limit: int
        :param niceness: New process niceness to assign to processes considered "hogs". Defaults to 10.
        :type niceness: int

        """
        cpu_hogs = self.get_cpu_hogs(limit, serialize=False)
        for proc in cpu_hogs:

            proc.nice(niceness)

    def get_file_openers(limit: int = 20, serialize=False):
        """get_file_openers(limit, serialize)
        Gets processes who currently have a number of files open above the *limit*.

        :param limit: Threshold on files open, processes above which should be identified. Defaults to 20.
        :type limit: int

                    if serialize:
                        proc_over_limit.append(serialize_proc(proc))
                    else:
                        proc_over_limit.append(proc)

        return proc_over_limit

[docs]    def safe_battery_level(self, limit: range(0, 100) = None):
        """safe_battery_level(limit)
        Verifies that the device has a sufficient battery level or is plugged in.

        :param limit: Battery percentage required if on battery.
            Defaults to :any:`safe_battery_level_limit`
            (|safe_battery_level_limit_default|) if no value is passed.
        :type limit: int
        :return: Indicator whether the battery level should be a concern.\n
            **True** if the device has sufficient energy, or **False** if
            the device is below the indicated battery percentage and unplugged.
        :rtype: bool

        .. warning::

            Always returns **None** on devices that are always plugged in or have no battery sensors.

        """
        if not limit:
            limit = self.safe_battery_level_limit
        battery_stats = psutil.sensors_battery()
        if battery_stats.power_plugged:
            return True

        return bool(battery_stats.percent > limit)


[docs]    def safe_temperature_level(self, limit: range(0, 100) = None):
        """safe_temperature_level(limit)
        Verifies if all device components are below a certain
        percentage of their factory critical temperature levels
        and returns any components that aren't.

        :param limit: Percentage of factory critical temperature levels which is considered unsafe.
            Defaults to :any:`safe_temperature_level_limit`
            (|safe_temperature_level_limit_default|) if no value is passed.
        :type limit: int
        :return: Hardware components approaching critical temperatures.\n
            Empty if no components pass the *limit*.
        :rtype: dict

        """
        if not limit:
            limit = self.safe_temperature_level_limit
        unsafe_components = {}
        sensors = psutil.sensors_temperatures()
        for key in sensors:
            unsafe_rows = []
            for row in sensors[key]:
                if row.current > row.critical * limit/100:
                    unsafe_rows.append(row)
            if unsafe_rows:
                unsafe_components.update({key: unsafe_rows})

        return unsafe_components


[docs]    def safe_load_avg_level(self, limit: float = None,
                            period: range(0, 3) = None):
        """safe_load_avg_level(limit, period)
        Verifies that the CPU load average for the
        specified time period is above the specified threshold

        :param limit: Maximum acceptable load average.
            Defaults to :any:`safe_load_avg_limit`
            (|safe_load_avg_limit_default|) if no value is passed.
        :type limit: float
        :param period: Period for which to check the load average - 1 (0), 5 (1), or 15 (2) minutes.
            Defaults to :any:`safe_load_avg_period`
            (|safe_load_avg_period_default|) if no value is passed.
        :type period: 0-2
        :return: Indicator whether the CPU load average for the specified *period* is above the *limit*.\n
            Returns **True** if within the safety threshold, or **False** if above *limit*.
        :rtype: bool

        """
        if not limit:
            limit = self.safe_load_avg_limit
        if not period:
            period = self.safe_load_avg_period

        return bool(getloadavg()[period] < limit)
